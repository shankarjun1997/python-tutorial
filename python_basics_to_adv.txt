2.6 Iterators

Allow looping through objects.

    def __init__(self, low, high):
        self.current = low

        return self

    def __next__(self):
        if self.current > self.high:
            raise StopIteration
        else:
            self.current += 1
            return self.current - 1

    print(num)  # Output: 1 2 3 4 5
